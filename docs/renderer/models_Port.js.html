/**
 * @file Contains model of a switch port
 * @author Jonathan Weatherspoon
 */


/**
 * @class 
 * @classdesc Models a port on a switch in a stack unit
 */
class Port {
    /**
     * Create a port model
     * @constructor
     * @param {number} stackUnit - The unit this port is on
     * @param {number} module - The module this port is on
     * @param {number} portID - The ID of the port
     */
    constructor(stackUnit, portModule, portID) {
        this._unit = stackUnit;
        this._module = portModule;
        this._id = portID;
    }

    /**
     * @type {number}
     */
    get unit() {
        return this._unit;
    }

    /**
     * @type {number}
     */
    set unit(newUnit) {
        if(newUnit < 0) newUnit = 0;
        this._unit = newUnit;
    }

    /**
     * @type {number}
     */
    get module() {
        return this._module;
    }

    /**
     * @type {number}
     */
    set module(newModule) {
        if(newModule < 0) newModule = 0;
        this._module = newModule;
    }

    /**
     * @type {number}
     */
    get id() {
        return this._id;
    }

    /**
     * @type {number}
     */
    set id(newID) {
        if(newID < 0) newID = 0;
        else if(newID > 48) newID = 48;
        this._id = newID;
    }
    
}
